📨 5. Kafka Configuration
@Configuration
public class KafkaConfig {

    @Bean
    public ProducerFactory<String, EventOutcome> producerFactory() {
        Map<String, Object> config = new HashMap<>();
        config.put(ProducerConfig.BOOTSTRAP_SERVERS_CONFIG, "localhost:9092");
        config.put(ProducerConfig.KEY_SERIALIZER_CLASS_CONFIG, StringSerializer.class);
        config.put(ProducerConfig.VALUE_SERIALIZER_CLASS_CONFIG, JsonSerializer.class);
        return new DefaultKafkaProducerFactory<>(config);
    }

    @Bean
    public KafkaTemplate<String, EventOutcome> kafkaTemplate() {
        return new KafkaTemplate<>(producerFactory());
    }
}
📥 6. Kafka Consumer (Core Processing)
@Service
@RequiredArgsConstructor
@Slf4j
public class EventOutcomeConsumer {

    private final BetService betService;
    private final SettlementProducer settlementProducer;

    @KafkaListener(topics = "event-outcomes", groupId = "settlement-group")
    public void consume(EventOutcome outcome) {
        log.info("Received outcome: {}", outcome);

        List<Bet> bets = betService.findBetsByEventId(outcome.getEventId());

        List<Bet> winningBets = bets.stream()
                .filter(b -> b.getEventWinnerId().equals(outcome.getWinnerId()))
                .toList();

        settlementProducer.sendSettlements(winningBets);
    }
}

🚀 8. RocketMQ Producer (Settlement Publisher)

@NoArgsConstructor
public class BetSettlementMessage {
    private String betId;
    private String userId;

}
Producer
@Service
@RequiredArgsConstructor
public class SettlementProducer {

    private final RocketMQTemplate rocketMQTemplate;

    public void sendSettlements(List<Bet> bets) {
        bets.forEach(bet -> {
            BetSettlementMessage msg = new BetSettlementMessage(
                    bet.getBetId(),
                    bet.getUserId(),
                    bet.getBetAmount()
            );

            rocketMQTemplate.convertAndSend("bet-settlements", msg);
        });
    }
}

Outbox Service
@Service
@RequiredArgsConstructor
public class OutboxService {

    private final OutboxRepository repo;
    private final ObjectMapper mapper;

    @Transactional
    public void saveEvent(EventOutcome event) {
        OutboxEvent outbox = new OutboxEvent();
        outbox.setAggregateId(event.getEventId());
        outbox.setType("EVENT_OUTCOME");
        outbox.setPayload(write(event));
        repo.save(outbox);
    }

    private String write(Object o) {
        try { return mapper.writeValueAsString(o); }
        catch (Exception e) { throw new RuntimeException(e); }
    }
}
Outbox Publisher (Kafka Relay)
@Component
@RequiredArgsConstructor
public class OutboxPublisher {

    private final OutboxRepository repo;
    private final KafkaTemplate<String, String> kafka;

    @Scheduled(fixedDelay = 2000)
    public void publish() {
        List<OutboxEvent> events = repo.findByPublishedFalse();

        for (OutboxEvent e : events) {
            kafka.send("event-outcomes", e.getAggregateId(), e.getPayload());
            e.setPublished(true);
            repo.save(e);
        }
    }
}

Kafka Consumer → Writes to Settlement Outbox
@Service
@RequiredArgsConstructor
public class OutcomeConsumer {

    private final BetRepository betRepo;
    private final SettlementOutboxService outbox;

    @KafkaListener(topics = "event-outcomes")
    public void consume(String payload) throws Exception {

        EventOutcome event = new ObjectMapper()
                .readValue(payload, EventOutcome.class);

        List<Bet> bets = betRepo.findByEventId(event.getEventId());

        bets.stream()
            .filter(b -> b.getEventWinnerId().equals(event.getWinnerId()))
            .forEach(outbox::saveSettlement);
    }
}

RocketMQ Publisher (Outbox Relay)
@Component
@RequiredArgsConstructor
public class SettlementPublisher {

    private final SettlementOutboxRepository repo;
    private final RocketMQTemplate rocket;

    @Scheduled(fixedDelay = 2000)
    public void publish() {
        List<SettlementOutbox> events = repo.findBySentFalse();

        for (SettlementOutbox e : events) {
            rocket.convertAndSend("bet-settlements", e.getPayload());
            e.setSent(true);
            repo.save(e);
        }
    }
}
